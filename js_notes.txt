function multiply(arr, n) { //recursion
    if (n <= 0) { //must have a base case, in this case it's the if statement
        return 1;
    } else {
        return multiply(arr, n - 1) * arr[n - 1];
    }
}
function countup(n) {
  if (n < 1) {
    return [];
  } else {
    const countArray = countup(n - 1);
    countArray.push(n);
    return countArray;
  }
}
console.log(countup(5)); // [ 1, 2, 3, 4, 5 ]
